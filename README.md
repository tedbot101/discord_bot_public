# music_bot.py 
py music_bot.py # launch command
command prefix '~'

# commands implemented
play : play music
play function also support queue
usage 1 : 
        ~play <url>

usage 2 : 
        ~play <song>

pause : pause the music playing
usage : 
        ~pause 

stop : stop the music 
usage : 
        ~stop

leave : tell the bot to leave the channel
usage : 
        ~leave
